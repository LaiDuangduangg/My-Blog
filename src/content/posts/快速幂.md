### 1. 核心代码模板

下面是快速幂的模板，注意别忘记取模：

```cpp
#include <iostream>
using namespace std;

typedef long long ll;

ll qpow(ll a, ll b, ll m) {
    ll res = 1;
    a %= m;
    while (b > 0) {
        if (b & 1) res = res * a % m;
        a = a * a % m;
        b >>= 1;
    }
    return res;
}

int main() {
    cout << qpow(2, 10, 1e9+7) << endl;
    return 0;
}
```
