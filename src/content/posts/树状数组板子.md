### 单点增加，区间查询
```cpp
struct Fenwick {
    int n;
    vector<long long> tree1;

    Fenwick(int n = 0) {
        init(n);
    }

    int lowbit(int x) {
        return x & -x;
    }

    void init(int _n) {
        n = _n;
        tree1.assign(n + 2, 0);
    }

    void add(vector<long long>& tree, int pos, long long val) {
        while (pos <= n) {
            tree[pos] += val;
            pos += lowbit(pos);
        }
    }

    long long ask(vector<long long>& tree, int x) {
        long long res = 0;
        while (x > 0) {
            res += tree[x];
            x -= lowbit(x);
        }
        return res;
    }

    void point_add(int pos, long long val) {
        add(tree1, pos, val);
    }

    long long prefix_sum(int x) {
        return ask(tree1, x);
    }

    long long range_query(int l, int r) {
        if (l > r) return 0;
        return prefix_sum(r) - prefix_sum(l - 1);
    }

    long long point_query(int pos) {
        return range_query(pos, pos);
    }
};
```

### 区间增加，区间查询
```cpp
struct Fenwick {
    int n;
    vector<long long> tree1;
    vector<long long> tree2;

    Fenwick(int n = 0) {
        init(n);
    }

    int lowbit(int x) {
        return x & -x;
    }

    void init(int _n) {
        n = _n;
        tree1.assign(n + 2, 0);
        tree2.assign(n + 2, 0);
    }

    void add(vector<long long>& tree, int pos, long long val) {
        while (pos <= n) {
            tree[pos] += val;
            pos += lowbit(pos);
        }
    }

    long long ask(vector<long long>& tree, int x) {
        long long res = 0;
        while (x > 0) {
            res += tree[x];
            x -= lowbit(x);
        }
        return res;
    }

    void range_add(int l, int r, long long val) {
        if (l > r) return;

        add(tree1, l, val);
        add(tree1, r + 1, -val);

        add(tree2, l, val * l);
        add(tree2, r + 1, -val * (r + 1));
    }

    long long range_prefix(int k) {
        if (k <= 0) return 0;
        return 1LL * (k + 1) * ask(tree1, k) - ask(tree2, k);
    }

    long long range_query(int l, int r) {
        if (l > r) return 0;
        return range_prefix(r) - range_prefix(l - 1);
    }

    long long point_query(int pos) {
        return range_query(pos, pos);
    }
};
```
